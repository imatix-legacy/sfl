unpack_date

#include "sfldate.h"
long
unpack_date (word packdate)

Synopsis

Converts a packed date back into a long value.

Source Code - (sfldate.c)

{
    int year;

    year = ((word) (packdate & 0xfe00) >> 9) + 80;
    return (MAKE_DATE (year > 80? 19: 20,
                       year,
                       (word) (packdate & 0x01e0) >> 5,
                       (word) (packdate & 0x001f)));
}
