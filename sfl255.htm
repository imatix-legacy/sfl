process_close_io

#include "sflproc.h"
void
process_close_io (PROCESS_DATA *procinfo)

Synopsis

Closes any file handles opened for the process. Use this function after you're finished with a process create full() call.

Source Code - (sflproc.c)

{
    if (procinfo-> in >= 0)
      {
        close (procinfo-> in);
        procinfo-> in = NULL_HANDLE;
      }
    if (procinfo-> out >= 0)
      {
        close (procinfo-> out);
        procinfo-> out = NULL_HANDLE;
      }
    if (procinfo-> err >= 0)
      {
        close (procinfo-> err);
        procinfo-> err = NULL_HANDLE;
      }
}
