strskp

#include "sflstr.h"
char *
strskp (
    const char *string)

Synopsis

Skips leading spaces in string, and returns a pointer to the first non-blank character. If this is a null, the end of the string was reached.

Source Code - (sflstr.c)

{
    char
        *skip = (char *) string;

    ASSERT (string);
    while (*skip == ' ')
        skip++;
    return (skip);
}
