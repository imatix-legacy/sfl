year_quarter

#include "sfldate.h"
int
year_quarter (long date)

Synopsis

Returns the year quarter, 1 to 4, depending on the month specified.

Source Code - (sfldate.c)

{
    return ((GET_MONTH (date) - 1) / 3 + 1);
}
