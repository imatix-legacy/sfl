xml_next_attr

#include "sflxml.h"
XML_ATTR *
xml_next_attr (XML_ATTR *attr)

Synopsis

Returns the next attribute following the specified attribute, or NULL if there are none.

Source Code - (sflxml.c)

{
    ASSERT (attr);

    if ((LIST *) attr-> next != & attr-> parent-> attrs)
        return attr-> next;
    else
        return NULL;
}
