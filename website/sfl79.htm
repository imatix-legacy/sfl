valid_time

#include "sfldate.h"
Bool
valid_time (long time)

Synopsis

Returns TRUE if the time is valid or zero; returns FALSE if the time is not valid.

Source Code - (sfldate.c)

{
    return (GET_SECOND (time) < 60
        &&  GET_MINUTE (time) < 60
        &&  GET_HOUR   (time) < 24);
}
