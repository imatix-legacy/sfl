tree_first

#include "sfltree.h"
void *tree_first (void *tree)

Synopsis

Finds and returns the first node in a (sub-)tree.

Source Code - (sfltree.c)

{
    TREE
       *current;

    if ((!tree)
    ||  (tree == TREE_NULL))
        return NULL;

    current = tree;
    while (current-> left != TREE_NULL)
        current = current-> left;

    return current;
}
