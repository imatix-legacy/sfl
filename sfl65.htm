unpack_time

#include "sfldate.h"
long
unpack_time (word packtime)

Synopsis

Converts a packed time back into a long value.

Source Code - (sfldate.c)

{
    return (MAKE_TIME ((word) (packtime & 0xf800) >> 11,
                       (word) (packtime & 0x07e0) >> 5,
                       (word) (packtime & 0x001f) << 1, 0));
}
